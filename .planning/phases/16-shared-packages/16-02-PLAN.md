    **Step 3: Switch use-standard-list.ts.**
    Replace `apps/portal/src/hooks/use-standard-list.ts` with a thin re-export file:
    ```typescript
    'use client';
    // Re-export from shared package - portal consumers import from here for backward compatibility
    export { useStandardList } from '@splits-network/shared-hooks';
    export type { UseStandardListOptions, UseStandardListReturn, FetchParams, FetchResponse } from '@splits-network/shared-hooks';

    // Re-export UI components (consumers import these from use-standard-list for convenience)
    export { SearchInput, PaginationControls, EmptyState, StandardListLoadingState, ErrorState, ViewModeToggle, MobileDetailOverlay } from '@splits-network/shared-ui';
    ```

    BUT WAIT: The shared-hooks version of use-standard-list no longer imports `useAuth` from Clerk. Portal consumers that relied on automatic auth need to pass `getToken` now. Check how use-standard-list is consumed in portal:
    - All 11 consumers (admin pages) use the hook and rely on automatic Clerk auth (`requireAuth: true` default).
    - Solution: Create a portal-specific wrapper hook in this file that auto-injects `getToken` from Clerk:
    ```typescript
    import { useAuth } from '@clerk/nextjs';
    import { useSearchParams, useRouter, usePathname } from 'next/navigation';
    import { useStandardList as useStandardListBase, type UseStandardListOptions, type UseStandardListReturn } from '@splits-network/shared-hooks';

    export function useStandardList<T = any, F extends Record<string, any> = Record<string, any>>(
        options: UseStandardListOptions<T, F>
    ): UseStandardListReturn<T, F> {
        const { getToken } = useAuth();
        const router = useRouter();
        const pathname = usePathname();
        const searchParams = useSearchParams();

        return useStandardListBase<T, F>({
            ...options,
            getToken: options.getToken ?? getToken,
            urlSync: options.urlSync ?? (options.syncToUrl !== false ? {
                searchParams: searchParams,
                pathname,
                replace: (url, opts) => router.replace(url, opts),
            } : undefined),
        });
    }
    ```
    This way, ALL existing portal consumers work with zero changes to their code. The admin app will create its own wrapper with its own Clerk instance.
