   ```typescript
   export interface CompanyPerk {
       company_id: string;

       perk?: Perk;

   export interface CompanyCultureTag {
       company_id: string;
       culture_tag_id: string;
       culture_tag?: CultureTag;

       skill_id: string;
       skill?: Skill;
   ```
