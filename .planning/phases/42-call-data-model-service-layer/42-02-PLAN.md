// Enums
export type CallStatus = 'scheduled' | 'active' | 'completed' | 'cancelled';
export type CallEntityType = 'application' | 'job' | 'company' | 'firm' | 'candidate';
export type CallParticipantRole = 'host' | 'participant' | 'observer';
export type RecordingStatus = 'pending' | 'recording' | 'processing' | 'ready' | 'failed';
export type TranscriptStatus = 'pending' | 'processing' | 'ready' | 'failed';
export type SummaryStatus = 'pending' | 'processing' | 'ready' | 'failed';

// Lookup
export interface CallType {
  slug: string;
  label: string;
  default_duration_minutes: number;
  requires_recording_consent: boolean;
  supports_ai_summary: boolean;
  created_at: string;
}

// Core
export interface Call {
  id: string;
  call_type: string;
  status: CallStatus;
  title: string | null;
  livekit_room_name: string | null;
  scheduled_at: string | null;
  started_at: string | null;
  ended_at: string | null;
  duration_minutes: number | null;
  created_by: string;
  deleted_at: string | null;
  created_at: string;
  updated_at: string;
}

export interface CallEntityLink {
  id: string;
  call_id: string;
  entity_type: CallEntityType;
  entity_id: string;
  created_at: string;
}

  id: string;
  call_id: string;
  user_id: string;
  role: CallParticipantRole;
  joined_at: string | null;
  left_at: string | null;
  consent_given: boolean;
  consent_at: string | null;
  created_at: string;
}

export interface CallAccessToken {
  id: string;
  call_id: string;
  user_id: string;
  token: string;
  expires_at: string;
  used_at: string | null;
  created_at: string;

// Artifacts
export interface CallRecording {
  id: string;
  call_id: string;
  recording_status: RecordingStatus;

  duration_seconds: number | null;
  file_size_bytes: number | null;

  <name>Task 2: Implement repository layer</name>
  <files>services/call-service/src/v2/repository.ts</files>
  <action>
Create `CallRepository` class following the InterviewRepository pattern (class with Supabase client, methods for each operation).

**Constructor:** Takes `supabaseUrl: string, supabaseKey: string`, creates Supabase client. Expose `getSupabase()` method.

**Call CRUD methods:**

1. `listCalls(params: StandardListParams, filters: CallListFilters)` -> `StandardListResponse<CallWithParticipants>`
   - Query `calls` table with soft-delete filter (`deleted_at IS NULL`)
   - Apply filters: `call_type`, `status`, date range on `scheduled_at`
   - If `entity_type` + `entity_id` provided: JOIN through `call_entity_links` to filter
   - JOIN `call_participants` to get participants with user details (first_name, last_name, avatar_url from users table)
   - JOIN `call_entity_links` to include entity links
   - Apply StandardListParams pagination (page, limit, sort_by, sort_order)
   - Return `{ data, pagination }` per StandardListResponse pattern
   - Use Supabase `.select()` with nested selects for joins: `*, call_participants(*, user:users(first_name, last_name, avatar_url, email)), call_entity_links(*)`
   - Count query for pagination total

2. `getCall(id: string)` -> `Call | null`
   - Simple by-ID fetch, filter `deleted_at IS NULL`

3. `getCallDetail(id: string, include?: string[])` -> `CallDetail | null`
   - Fetch call with participants and entity_links always included
   - If `include` contains 'recordings': join call_recordings
   - If `include` contains 'transcript': join call_transcripts (latest one)
   - If `include` contains 'summary': join call_summaries
   - If `include` contains 'notes': join call_notes with user details
   - Build select string dynamically based on include flags

4. `createCall(input: CreateCallInput, createdBy: string)` -> `Call`
   - Insert into calls table
   - Return the created call

5. `updateCall(id: string, input: UpdateCallInput)` -> `Call`
   - Update calls table by id, filter deleted_at IS NULL

6. `deleteCall(id: string)` -> `void`
   - Soft delete: SET deleted_at = now()

7. `updateCallStatus(id: string, status: CallStatus, extraFields?: Partial<Call>)` -> `Call`
   - Update status + optional fields (started_at, ended_at, duration_minutes, livekit_room_name)

**Participant methods:**

8. `addParticipant(callId: string, input: { user_id: string, role: CallParticipantRole })` -> `CallParticipant`
9. `removeParticipant(participantId: string)` -> `void` (hard delete)
10. `getCallParticipants(callId: string)` -> `CallParticipantWithUser[]`
11. `updateParticipantJoin(participantId: string, joinedAt: string)` -> `CallParticipant`
12. `updateParticipantLeave(participantId: string, leftAt: string)` -> `CallParticipant`
13. `updateParticipantConsent(participantId: string, consentGiven: boolean)` -> `CallParticipant`

**Entity link methods:**

14. `addEntityLink(callId: string, input: { entity_type: CallEntityType, entity_id: string })` -> `CallEntityLink`
15. `removeEntityLink(linkId: string)` -> `void`
16. `getCallEntityLinks(callId: string)` -> `CallEntityLink[]`

**Access token methods:**

17. `createAccessToken(input: { call_id: string, user_id: string, token: string, expires_at: string })` -> `CallAccessToken`
18. `getAccessTokenByToken(token: string)` -> `CallAccessToken | null`
19. `markTokenUsed(tokenId: string)` -> `void`

**Helper:**

20. `resolveUserId(clerkUserId: string)` -> `string | null`
    - Query users table by clerk_user_id, return id. Follow video-service InterviewRepository pattern.

All methods use `{ data, error }` Supabase pattern. Throw on error with statusCode attached (404 for not found, etc).

Keep the file focused — if it exceeds ~200 lines, split into `repository.ts` (calls CRUD), `participant-repository.ts`, `artifact-repository.ts`. Use your judgment on the split point, but prefer keeping it in one file if under 250 lines since the methods are short.
